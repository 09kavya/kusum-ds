27. Display the string on the output screen but ask the text alignment option given by the user (1 for left, 2 for center, 3 for right).

Code:

        #include 
            #include 
            
            #define MAX_LENGTH 500
            #define TOTAL_WIDTH 50 // Width of the output area
            
            void printLeftAligned(const char *str) {
                printf("%-*s\n", TOTAL_WIDTH, str); // Left alignment
            }
            
            void printCenterAligned(const char *str) {
                int len = strlen(str);
                int padding = (TOTAL_WIDTH - len) / 2; // Calculate padding for center alignment
                printf("%*s%s\n", padding, "", str); // Center alignment
            }
            
            void printRightAligned(const char *str) {
                printf("%*s\n", TOTAL_WIDTH, str); // Right alignment
            }
            
            int main() {
                char str[MAX_LENGTH];
                int alignmentOption;
            
                printf("Enter a string: ");
                fgets(str, sizeof(str), stdin);
                str[strcspn(str, "\n")] = 0; // Remove the newline character
            
                printf("Choose text alignment option:\n");
                printf("1. Left Align\n");
                printf("2. Center Align\n");
                printf("3. Right Align\n");
                printf("Enter your choice (1/2/3): ");
                scanf("%d", &alignmentOption);
            
                switch (alignmentOption) {
                    case 1:
                        printLeftAligned(str);
                        break;
                    case 2:
                        printCenterAligned(str);
                        break;
                    case 3:
                        printRightAligned(str);
                        break;
                    default:
                        printf("Invalid option selected.\n");
                        break;
                }
            
                return 0;
            }
