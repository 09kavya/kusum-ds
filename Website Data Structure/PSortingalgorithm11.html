            #define MAX_SIZE 100 
            #define MAX_LENGTH 100 
            void swap(char* a, char* b) {
                char temp[MAX_LENGTH];
                strcpy(temp, a);
                strcpy(a, b);
                strcpy(b, temp);
            }
            
            void selectionSort(char arr[][MAX_LENGTH], int n) {
                for (int i = 0; i < n - 1; i++) {
                    int minIdx = i;
                    for (int j = i + 1; j < n; j++) {
                        if (strcmp(arr[j], arr[minIdx]) < 0) {
                            minIdx = j;
                        }
                    }
                    swap(arr[i], arr[minIdx]);
                }
            }
            
            void printArray(char arr[][MAX_LENGTH], int n) {
                for (int i = 0; i < n; i++) {
                    printf("%s\n", arr[i]);
                }
                printf("\n");
            }
            
            int main() {
                int n;
                printf("Enter the number of strings (max %d): ", MAX_SIZE);
                scanf("%d", &n);

                if (n > MAX_SIZE) {
                    printf("Error: Maximum size exceeded.\n");
                    return 1;
                }
            
                char arr[MAX_SIZE][MAX_LENGTH];
            
                printf("Enter %d strings:\n", n);
                for (int i = 0; i < n; i++) {
                    fgets(arr[i], MAX_LENGTH, stdin);
                    arr[i][strcspn(arr[i], "\n")] = 0; 
                }
            
                clock_t start = clock();
                selectionSort(arr, n);
                clock_t end = clock();
            
                double time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
