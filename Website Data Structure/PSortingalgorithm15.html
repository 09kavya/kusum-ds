            #define MAX_SIZE 100 
            #define MAX_LENGTH 100 
            void insertionSort(char arr[][MAX_LENGTH], int n) {
                for (int i = 1; i < n; i++) {
                    char key[MAX_LENGTH];
                    strcpy(key, arr[i]); 
                    int j = i - 1;5
                    while (j >= 0 && strcmp(arr[j], key) > 0) {
                        strcpy(arr[j + 1], arr[j]);
                        j--;
                    }
                    strcpy(arr[j + 1], key); 
                }
            }
            
            void printArray(char arr[][MAX_LENGTH], int n) {
                for (int i = 0; i < n; i++) {
                    printf("%s\n", arr[i]);
                }
                printf("\n");
            }
            
            int main() {
                int n;
                printf("Enter the number of strings (max %d): ", MAX_SIZE);

                char arr[MAX_SIZE][MAX_LENGTH];
            
                printf("Enter %d strings:\n", n);
                for (int i = 0; i < n; i++) {
                    gets(arr[i]); 
                }
            
                clock_t start = clock();
                insertionSort(arr, n);
                clock_t end = clock();
            
                double time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
            
                printf("Sorted strings:\n");
                printArray(arr, n);
                printf("Time taken for insertion sort: %f seconds\n", time_taken);
