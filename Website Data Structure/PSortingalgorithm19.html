            #define MAX_SIZE 100

            void swap(char* a, char* b) {
                char temp[MAX_LENGTH];
                strcpy(temp, a);
                strcpy(a, b);
                strcpy(b, temp);
            }
            
            void iterativeQuickSort(char arr[][MAX_LENGTH], int n) {
                int stack[MAX_SIZE];
                int top = -1; 
                stack[++top] = 0;
                stack[++top] = n - 1;
            
                while (top >= 0) {
                    // Pop high and low
                    int high = stack[top--];
                    int low = stack[top--];
            
                    char pivot[MAX_LENGTH];
                    strcpy(pivot, arr[high]);
                    int i = low - 1;
            
                    for (int j = low; j < high; j++) {
                        if (strcmp(arr[j], pivot) <= 0) {
                            i++;
                            swap(arr[i], arr[j]);
                        }
                    }
                    swap(arr[i + 1], arr[high]); 
                    int pi = i + 1;
            
                  
                    if (pi - 1 > low) {
                        stack[++top] = low;
                        stack[++top] = pi - 1;
                    }
                  
                    if (pi + 1 < high) {
                        stack[++top] = pi + 1;
                        stack[++top] = high;
                    }
                }
            }
            
            void printArray(char arr[][MAX_LENGTH], int n) {
                for (int i = 0; i < n; i++) {
                    printf("%s\n", arr[i]);
                }
                printf("\n");
            }
