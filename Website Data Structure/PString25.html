Code:

        #include 

            #define MAX_LENGTH 500
            
            void trimSpaces(char *str, char *trimmed) {
                int start = 0, end = 0, i = 0;
            
                while (str[start] == ' ') {
                    start++;
                }
            
              
                end = start;
                while (str[end] != '\0') {
                    end++;
                }
                end--; 
                while (end >= start && str[end] == ' ') {
                    end--;
                }
            
               
                for (i = 0; start <= end; i++, start++) {
                    trimmed[i] = str[start];
                }
                trimmed[i] = '\0'; 
            }
            
            int main() {
                char str[MAX_LENGTH];
                char trimmed[MAX_LENGTH];
                printf("Enter a string: ");
                gets(str);
                str[strcspn(str, "\n")] = 0;
                trimSpaces(str, trimmed);
                printf("Trimmed string: '%s'\n", trimmed);
                return 0;
            }
