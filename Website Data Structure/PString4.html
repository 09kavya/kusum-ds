4. Enter the string from user and convert into title case.

Code:

        #include 

            int main() {
                char string[100]; 
            
                printf("Enter a string: ");
                gets(string); 
            
               
                if (string[0] >= 'a' && string[0] <= 'z') {
                    string[0] = string[0] - 32; // Capitalize the first letter
                }
            
                
                for (int i = 1; string[i] != '\0'; i++) {
                    // Check if the current character is a space
                    if (string[i - 1] == ' ' && string[i] >= 'a' && string[i] <= 'z') {
                        string[i] = string[i] - 32; // Capitalize the letter after a space
                    } else if (string[i] >= 'A' && string[i] <= 'Z') {
                        string[i] = string[i] + 32; // Convert other uppercase letters to lowercase
                    }
                }
            
                printf("Title case string: %s\n", string); 
            
                return 0;
            }
