3. Create a stack implementing the operations on an integer nodes (dynamically) (push, pop, peek, isempty, isfull)

Code:

        #include 
#include 

typedef struct Node {
    int data;
    struct Node* next;
} Node;

typedef struct Stack {
    Node* top;
    int maxSize;
    int currentSize;
} Stack;

Stack* createStack(int maxSize) {
    Stack* stack = (Stack*)malloc(sizeof(Stack));
    stack->top = NULL;
    stack->maxSize = maxSize;
    stack->currentSize = 0;
    return stack;
}

int isFull(Stack* stack) {
    return stack->currentSize == stack->maxSize;
}

int isEmpty(Stack* stack) {
    return stack->currentSize == 0;
}

void push(Stack* stack, int value) {
    if (isFull(stack)) {
        printf("Stack is full, cannot push %d\n", value);
        return;
    }
    Node* newNode = (Node*)malloc(sizeof(Node));
    newNode->data = value;
    newNode->next = stack->top;
    stack->top = newNode;
    stack->currentSize++;
    printf("%d pushed onto stack.\n", value);
}

int pop(Stack* stack) {
    if (isEmpty(stack)) {
        printf("Stack underflow: no element to pop.\n");
        return -1; // Indicate underflow
    }
    Node* temp = stack->top;
    int poppedValue = temp->data;
    stack->top = stack->top->next;
    free(temp);
    stack->currentSize--;
    printf("%d popped from the stack.\n", poppedValue);
    return poppedValue;
}

int peek(Stack* stack) {
    if (isEmpty(stack)) {
        printf("Stack is empty: no elements to peek.\n");
        return -1; // Indicate empty
    }
    return stack->top->data;
}

void freeStack(Stack* stack) {
    while (!isEmpty(stack)) {
        pop(stack);
    }
    free(stack);
}

int main() {
    Stack* stack = createStack(5); // Create a stack with a maximum size of 5
    int choice, value;

    while (1) {
        printf("1. Push\n2. Pop\n3. Peek\n4. Check if empty\n5. Check if full\n6. Exit: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter the integer value: ");
                scanf("%d", &value);
                push(stack, value);
                break;
            case 2:
                pop(stack);
                break;
            case 3:
                value = peek(stack);
                if (value != -1) {
                    printf("Top element is: %d\n", value);
                }
                break;
            case 4:
                if (isEmpty(stack)) {
                    printf("Stack is empty.\n");
                } else {
                    printf("Stack is not empty.\n");
                }
                break;
            case 5:
                if (isFull(stack)) {
                    printf("Stack is full.\n");
                } else {
                    printf("Stack is not full.\n");
                }
                break;
            case 6:
                freeStack(stack);
                return 0;
            default:
                printf("Invalid choice. Please try again.\n");
        }
    }
}
