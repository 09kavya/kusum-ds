14. Enter a string array from the user and perform bubble sort algorithm through recursion. Also calculate the worst and best complexity with exact time.

            #define MAX_SIZE 100 
            #define MAX_LENGTH 100 
            void swap(char* a, char* b) {
                char temp[MAX_LENGTH];
                strcpy(temp, a);
                strcpy(a, b);
                strcpy(b, temp);
            }
            
            void bubbleSortRecursive(char arr[][MAX_LENGTH], int n, int index) {
               
                if (n == 1) return;
                if (index == n - 1) {
                    bubbleSortRecursive(arr, n - 1, 0);
                    return;
                }
                if (strcmp(arr[index], arr[index + 1]) > 0) {
                    swap(arr[index], arr[index + 1]);
                }
            
                bubbleSortRecursive(arr, n, index + 1);
            }
            
            void printArray(char arr[][MAX_LENGTH], int n) {
                for (int i = 0; i < n; i++) {
                    printf("%s\n", arr[i]);
                }
                printf("\n");
            }
            
            int main() {
                int n;
            
                printf("Enter the number of strings (max %d): ", MAX_SIZE);
                scanf("%d", &n);
                getchar(); 
                if (n > MAX_SIZE) {
                    printf("Error: Maximum size exceeded.\n");
                    return 1;
                }
            
                char arr[MAX_SIZE][MAX_LENGTH];
                printf("Enter %d strings:\n", n);
                for (int i = 0; i < n; i++) {
                    gets(arr[i]); 
                }
                clock_t start = clock();
                bubbleSortRecursive(arr, n, 0);
                clock_t end = clock();
            
                double time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
            
                printf("Sorted strings:\n");
                printArray(arr, n);
                printf("Time taken for recursive bubble sort: %f seconds\n", time_taken);
            
                return 0;
            }
