3. Calculate the sum of natural numbers till 10 using recursion.

Code:

        #include < stdio.h >

        int sumOfNaturalNumbers(int n) {
            if (n == 0) 
                return 0;
            else
                return n + sumOfNaturalNumbers(n - 1); 
            }
                            
            int main() {
                int n = 10;
                printf("Sum of natural numbers up to %d is %d\n",
                 n, sumOfNaturalNumbers(n));
                return 0;
            }

Output:

        Sum of natural numbers up to 10 is 55
