20. Enter the string from the user and count the number of characters, words, spaces, and lines.

Code:

        #include 

            int main() { 
                char str[30]; 
                int i = 0, count = 0;
            
                printf("Enter the string: "); 
                gets(str); 
                while (str[i] != '\0') {  
                   
                    if (str[i] == ' ' || str[i] == '\t' || str[i] == '\n') {
                        count++;
                    }
                    i++;
                }
            
              
                printf("Number of whitespace characters: %d\n", count);
            
                return 0;
            }
