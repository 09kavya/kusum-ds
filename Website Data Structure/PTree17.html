            struct Node {
                int data;
                struct Node* left;
                struct Node* right;
            };
            struct Node* newNode(int data) {
                struct Node* node = (struct Node*)malloc(sizeof(struct Node));
                node->data = data;
                node->left = NULL;
                node->right = NULL;
                return node;
            }
            
            struct Node* insert(struct Node* root, int data) {
                if (root == NULL) {
                    return newNode(data);
                }
            
              
                if (root->left == NULL) {
                    root->left = insert(root->left, data);
                } else if (root->right == NULL) {
                    root->right = insert(root->right, data);
                } else {
                    printf("Cannot insert %d, as it would violate the strictly binary tree property.\n", data);
                }
                return root;
            }
            
            void inorder(struct Node* root) {
                if (root != NULL) {
                    inorder(root->left);
                    printf("%d ", root->data);
                    inorder(root->right);
                }
            }
            
            void freeTree(struct Node* root) {
                if (root != NULL) {
                    freeTree(root->left);
                    freeTree(root->right);
                    free(root);
                }
            }
            
            int main() {
                struct Node* root = NULL;
                int value, n;
            
                printf("Enter the number of nodes for the strictly binary tree: ");
                scanf("%d", &n);
            
                printf("Enter the values for the nodes:\n");
                for (int i = 0; i < n; i++) {
                    printf("Node %d: ", i + 1);
                    scanf("%d", &value);
                    root = insert(root, value);
                }
            
                printf("Inorder traversal of the constructed strictly binary tree: ");
                inorder(root);
                printf("\n");
            
                freeTree(root);
                
                return 0;
            }
