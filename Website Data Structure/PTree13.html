13. Find kth smallest element in a binary search tree through array.

Code:

            struct Node {
                int data;
                struct Node* left;
                struct Node* right;
            };
            
            struct Node* newNode(int data) {
                struct Node* node = (struct Node*)malloc(sizeof(struct Node));
                node->data = data;
                node->left = NULL;
                node->right = NULL;
                return node;
            }
            
            void storeInorder(struct Node* root, int* arr, int* index) {
                if (root == NULL) return;
            
                storeInorder(root->left, arr, index);
                arr[(*index)++] = root->data; 
                storeInorder(root->right, arr, index);
            }
            
            int kthSmallest(struct Node* root, int k) {
             
                int* arr = (int*)malloc(100 * sizeof(int));
                int index = 0;
                
                storeInorder(root, arr, &index);
            
                if (k > index || k < 1) {
                    free(arr);
                    return -1; 
                }
            
               
                int kthSmallestElement = arr[k - 1];
            
                free(arr); 
                return kthSmallestElement;
            }
            
            void inorder(struct Node* root) {
                if (root != NULL) {
                    inorder(root->left);
                    printf("%d ", root->data);
                    inorder(root->right);
                }
            }
            
            int main() {
                struct Node* root = newNode(50);
                root->left = newNode(30);
                root->right = newNode(70);
                root->left->left = newNode(20);
                root->left->right = newNode(40);
                root->right->left = newNode(60);
                root->right->right = newNode(80);
            
                printf("Inorder traversal of the BST: ");
                inorder(root);
                printf("\n");
            
                int k = 3; 
                int result = kthSmallest(root, k);
                if (result != -1) {
                    printf("The %dth smallest element in the BST is: %d\n", k, result);
                } else {
                    printf("The value of k is out of bounds.\n");
                }
            
                return 0;
            }
