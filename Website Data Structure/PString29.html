29. Enter the string from user and convert characters into their ASCII values in the integer array (for example, take 10 characters).

Code:

        #include 

            int main() {
                char string[11];  
                int asciiValues[10];  
            
                printf("Enter a string (up to 10 characters): ");
               gets(string);
            
               
                for (int i = 0; i < 10 && string[i] != '\0'; i++) {
                    asciiValues[i] = (int)string[i]; 
                }
            
                printf("ASCII values:\n");
                for (int i = 0; i < 10 && string[i] != '\0'; i++) {
                    printf("Character: '%c' : ASCII: %d\n", string[i], asciiValues[i]);
                }
            
                return 0;
            }
