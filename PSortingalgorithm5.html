5. Enter a integer array from the user and perform insertion sort algorithm through iteration. Also calculate the worst and best complexity with exact time.

        #include 
            #include 
            
            void bubbleSortRecursive(int arr[], int n) {
                if (n == 1)
                return;
                for (int i = 0; i < n - 1; i++) {
                    if (arr[i] > arr[i + 1]) {
                        int temp = arr[i];
                        arr[i] = arr[i + 1];
                        arr[i + 1] = temp;
                    }
                }
                bubbleSortRecursive(arr, n - 1);
            }
            
            void printArray(int arr[], int n) {
                for (int i = 0; i < n; i++) {
                    printf("%d ", arr[i]);
                }
                printf("\n");
            }
            
            int main() {
                int n;
                printf("Enter the number of elements: ");
                scanf("%d", &n);
                int arr[n];
                printf("Enter %d integers:\n", n);
                for (int i = 0; i < n; i++) {
                    scanf("%d", &arr[i]);
                }
            
                // Measure time for bubble sort (worst case)
                clock_t start = clock();
                bubbleSortRecursive(arr, n);
                clock_t end = clock();
            
                double time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
            
                printf("Sorted array:\n");
                printArray(arr, n);
                printf("Time taken for bubble sort (worst case): %f seconds\n", time_taken);
            
                // Best case: sorted array
                // Reset the array for best-case scenario
                printf("Enter %d integers for best case (sorted input):\n", n);
                for (int i = 0; i < n; i++) {
                    arr[i] = i; // Create a sorted array
                }
            
                start = clock();
                bubbleSortRecursive(arr, n);
                end = clock();
            
                time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
            
                printf("Sorted array (best case):\n");
                printArray(arr, n);
                printf("Time taken for bubble sort (best case): %f seconds\n", time_taken);
            
                return 0;
            }
