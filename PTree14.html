            struct Node {
                int data;
                struct Node* left;
                struct Node* right;
            };
            
            struct Node* newNode(int data) {
                struct Node* node = (struct Node*)malloc(sizeof(struct Node));
                node->data = data;
                node->left = NULL;
                node->right = NULL;
                return node;
            }
            
            struct Node* insert(struct Node* root, int data) {
                if (root == NULL) {
                    return newNode(data);
                }
                if (data < root->data) {
                    root->left = insert(root->left, data);
                } else {
                    root->right = insert(root->right, data);
                }
                return root;
            }
            
            struct Node* buildBSTFromPreorder(int* preorder, int* index, int size) {
                if (*index >= size) {
                    return NULL;
                }
            
                struct Node* root = newNode(preorder[*index]);
                (*index)++;
            
                if (*index < size && preorder[*index] < root->data) {
                    root->left = buildBSTFromPreorder(preorder, index, size);
                }
                if (*index < size && preorder[*index] > root->data) {
                    root->right = buildBSTFromPreorder(preorder, index, size);
                }
            
                return root;
            }
            
            void postorder(struct Node* root) {
                if (root == NULL) return;
            
                postorder(root->left);
                postorder(root->right);
                printf("%d ", root->data);
            }
