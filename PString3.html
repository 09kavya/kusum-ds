3. Enter the string from user and convert into sentence case.

Code:

        #include 

            int main() {
                char string[100]; 
            
                printf("Enter a string: ");
                gets(string); 
            
                if (string[0] >= 'a' && string[0] <= 'z') {
                    string[0] = string[0] - 32; // Convert first character to uppercase
                }
            
                // Convert the rest of the string to lowercase
                for (int i = 1; string[i] != '\0'; i++) {
                    if (string[i] >= 'A' && string[i] <= 'Z') {
                        string[i] = string[i] + 32; 
                    }
                }
            
                printf("Sentence case string: %s\n", string); 
            
                return 0;
            }
