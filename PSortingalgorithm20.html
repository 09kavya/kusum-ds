20. Enter a string array from the user and perform quick sort algorithm through recursion. Also calculate the worst and best complexity with exact time.

            #define MAX_SIZE 100 
            #define MAX_LENGTH 100 
            void swap(char* a, char* b) {
                char temp[MAX_LENGTH];
                strcpy(temp, a);
                strcpy(a, b);
                strcpy(b, temp);
            }
            
            int partition(char arr[][MAX_LENGTH], int low, int high) {
                char pivot[MAX_LENGTH];
                strcpy(pivot, arr[high]); 
                int i = low - 1; 
            
                for (int j = low; j < high; j++) {
                    
                    if (strcmp(arr[j], pivot) <= 0) {
                        i++;
                        swap(arr[i], arr[j]); 
                    }
                }
                swap(arr[i + 1], arr[high]); 
                return (i + 1);
            }
            
            void quickSort(char arr[][MAX_LENGTH], int low, int high) {
                if (low < high) {
                    int pi = partition(arr, low, high); 
                    quickSort(arr, low, pi - 1); 
                    quickSort(arr, pi + 1, high); 
                }
            }
            
            void printArray(char arr[][MAX_LENGTH], int n) {
                for (int i = 0; i < n; i++) {
                    printf("%s\n", arr[i]);
                }
                printf("\n");
            }
            
            int main() {
                int n;
                printf("Enter the number of strings (max %d): ", MAX_SIZE);
                scanf("%d", &n);
                getchar(); 
                if (n > MAX_SIZE) {
                    printf("Error: Maximum size exceeded.\n");
                    return 1;
                }
            
                char arr[MAX_SIZE][MAX_LENGTH];
            
                printf("Enter %d strings:\n", n);
                for (int i = 0; i < n; i++) {
                    gets(arr[i]); 
                }
            
                clock_t start = clock();
                quickSort(arr, 0, n - 1);
                clock_t end = clock();
            
                double time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
            
                printf("Sorted strings:\n");
                printArray(arr, n);
                printf("Time taken for recursive quick sort: %f seconds\n", time_taken);
            
                return 0;
            }
