            #define MAX_SIZE 100 
            #define MAX_LENGTH 100 
            
            
            void insert(char arr[][MAX_LENGTH], int n) {
               
                if (n <= 1) return;
            
                insert(arr, n - 1);
            
                char last[MAX_LENGTH];
                strcpy(last, arr[n - 1]); 
                int j = n - 2;
            
                while (j >= 0 && strcmp(arr[j], last) > 0) {
                    strcpy(arr[j + 1], arr[j]);
                    j--;
                }
                strcpy(arr[j + 1], last); 
            }
            
            void printArray(char arr[][MAX_LENGTH], int n) {
                for (int i = 0; i < n; i++) {
                    printf("%s\n", arr[i]);
                }
                printf("\n");
            }
            
            int main() {
                int n;
                printf("Enter the number of strings (max %d): ", MAX_SIZE);
                scanf("%d", &n);
                getchar(); 
                if (n > MAX_SIZE) {
                    printf("Error: Maximum size exceeded.\n");
                    return 1;
                }
            
                char arr[MAX_SIZE][MAX_LENGTH];
                printf("Enter %d strings:\n", n);
                for (int i = 0; i < n; i++) {
                    gets(arr[i]); 
                }
                clock_t start = clock();
                insert(arr, n);
                clock_t end = clock();
            
                double time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
            
                printf("Sorted strings:\n");
                printArray(arr, n);
                printf("Time taken for recursive insertion sort: %f seconds\n", time_taken);
            
                return 0;
            }
