            struct Node {
                int data;
                struct Node* left;
                struct Node* right;
            };
            
            struct Node* newNode(int data) {
                struct Node* node = (struct Node*)malloc(sizeof(struct Node));
                node->data = data;
                node->left = NULL;
                node->right = NULL;
                return node;
            }
            
            struct Node* buildBSTFromPreorder(int* preorder, int* index, int size, int min, int max) {
                if (*index >= size) return NULL;
            
                int value = preorder[*index];
            
                if (value < min || value > max) return NULL;
            
                struct Node* root = newNode(value);
                (*index)++; 
                root->left = buildBSTFromPreorder(preorder, index, size, min, value);
                root->right = buildBSTFromPreorder(preorder, index, size, value, max);
            
                return root;
            }
            
            void inorder(struct Node* root) {
                if (root != NULL) {
                    inorder(root->left);
                    printf("%d ", root->data);
                    inorder(root->right);
                }
            }
            
            int main() {
                int preorder[] = {50, 30, 20, 40, 70, 60, 80};
                int size = sizeof(preorder) / sizeof(preorder[0]);
                int index = 0;
            
                struct Node* root = buildBSTFromPreorder(preorder, &index, size, INT_MIN, INT_MAX);
            
                printf("Inorder traversal of the constructed BST: ");
                inorder(root);
                printf("\n");
            
                return 0;
