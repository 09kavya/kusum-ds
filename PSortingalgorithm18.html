            #define MAX_SIZE 100

            void merge(char arr[][MAX_LENGTH], int left, int mid, int right) {
                int i, j, k;
                int n1 = mid - left + 1;
                int n2 = right - mid;
            
                char L[MAX_SIZE][MAX_LENGTH], R[MAX_SIZE][MAX_LENGTH];
            
                for (i = 0; i < n1; i++)
                    strcpy(L[i], arr[left + i]);
                for (j = 0; j < n2; j++)
                    strcpy(R[j], arr[mid + 1 + j]);
            
             
                i = 0; 
                j = 0; 
                k = left; 
            
                while (i < n1 && j < n2) {
                    if (strcmp(L[i], R[j]) <= 0) {
                        strcpy(arr[k], L[i]);
                        i++;
                    } else {
                        strcpy(arr[k], R[j]);
                        j++;
                    }
                    k++;
                }
                while (i < n1) {
                    strcpy(arr[k], L[i]);
                    i++;
                    k++;
                }
            
                while (j < n2) {
                    strcpy(arr[k], R[j]);
                    j++;
                    k++;
                }
            }
            void mergeSort(char arr[][MAX_LENGTH], int left, int right) {
                if (left < right) {
                    int mid = left + (right - left) / 2; 
                    mergeSort(arr, left, mid);
                    mergeSort(arr, mid + 1, right);
                    merge(arr, left, mid, right);
                }
            }
            void printArray(char arr[][MAX_LENGTH], int n) {
                for (int i = 0; i < n; i++) {
