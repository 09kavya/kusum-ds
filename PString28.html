28. Create a dictionary having 10 words and their meaning with the following operations: [Create, Delete, Search].

Code:

        #include 
            #include 
            
            #define MAX_WORDS 10
            #define MAX_WORD_LENGTH 50
            #define MAX_MEANING_LENGTH 200
            
            typedef struct {
                char word[MAX_WORD_LENGTH];
                char meaning[MAX_MEANING_LENGTH];
            } DictionaryEntry;
            
            void createEntry(DictionaryEntry dict[], int *count) {
                if (*count >= MAX_WORDS) {
                    printf("Dictionary is full. Cannot add more words.\n");
                    return;
                }
            
                printf("Enter word: ");
                scanf("%s", dict[*count].word);
                printf("Enter meaning: ");
                scanf(" %[^\n]", dict[*count].meaning); 
            
                (*count)++;
                printf("Entry added successfully!\n");
            }
            
            
            void deleteEntry(DictionaryEntry dict[], int *count) {
                char word[MAX_WORD_LENGTH];
                printf("Enter the word to delete: ");
                scanf("%s", word);
            
                for (int i = 0; i < *count; i++) {
                    if (strcmp(dict[i].word, word) == 0) {
                   
                        for (int j = i; j < *count - 1; j++) {
                            dict[j] = dict[j + 1];
                        }
                        (*count)--; // Decrease the count
                        printf("Entry deleted successfully!\n");
                        return;
                    }
                }
                printf("Word not found in the dictionary.\n");
            }
            
            
            void searchEntry(DictionaryEntry dict[], int count) {
                char word[MAX_WORD_LENGTH];
                printf("Enter the word to search: ");
                scanf("%s", word);
            
                for (int i = 0; i < count; i++) {
                    if (strcmp(dict[i].word, word) == 0) {
                        printf("Word: %s\nMeaning: %s\n", dict[i].word, dict[i].meaning);
                        return;
                    }
                }
                printf("Word not found in the dictionary.\n");
            }
            
            
            void displayEntries(DictionaryEntry dict[], int count) {
                if (count == 0) {
                    printf("The dictionary is empty.\n");
                    return;
                }
                printf("\nDictionary Entries:\n");
                for (int i = 0; i < count; i++) {
                    printf("%s: %s\n", dict[i].word, dict[i].meaning);
                }
            }
            
            int main() {
                DictionaryEntry dictionary[MAX_WORDS] = {
                    {"apple", "A fruit that is usually red, green, or yellow."},
                    {"banana", "A long yellow fruit."},
                    {"grape", "A small round fruit, usually purple or green."},
                    {"orange", "A citrus fruit that is orange in color."},
                    {"strawberry", "A red fruit with seeds on its surface."},
                    {"kiwi", "A small, brown, fuzzy fruit."},
                    {"mango", "A tropical stone fruit."},
                    {"pineapple", "A large tropical fruit with spiky skin."},
                    {"pear", "A green or yellow fruit that is sweet and juicy."},
                    {"peach", "A soft, round fruit with fuzzy skin."}
                };
                
                int count = 10; 
                int choice;
            
                do {
                    printf("\nDictionary Operations:\n");
                    printf("1. Create Entry\n");
                    printf("2. Delete Entry\n");
                    printf("3. Search Entry\n");
                    printf("4. Display All Entries\n");
                    printf("5. Exit\n");
                    printf("Enter your choice: ");
                    scanf("%d", &choice);
            
                    switch (choice) {
                        case 1:
                            createEntry(dictionary, &count);
                            break;
                        case 2:
                            deleteEntry(dictionary, &count);
                            break;
                        case 3:
                            searchEntry(dictionary, count);
                            break;
                        case 4:
                            displayEntries(dictionary, count);
                            break;
                        case 5:
                            printf("Exiting the program.\n");
                            break;
                        default:
                            printf("Invalid choice. Please try again.\n");
                            break;
                    }
                } while (choice != 5);
            
                return 0;
            }
