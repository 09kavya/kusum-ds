            #define MAX 100 
            
            
            void push(char stack[], int *top, char value) {
                if (*top < MAX - 1) {
                    stack[++(*top)] = value;
                } else {
                    printf("Stack overflow: cannot push '%c'\n", value);
                }
            }
            
            
            char pop(char stack[], int *top) {
                if (*top >= 0) {
                    return stack[(*top)--];
                } else {
                    return '\0'; 
                }
            }
            int isBalanced(const char *expr) {
                char stack[MAX];
                int top = -1; 
            
                for (int i = 0; expr[i] != '\0'; i++) {
                    char current = expr[i];
            
                  
                    if (current == '(') {
                        push(stack, &top, current);
                    }
                   
                    else if (current == ')') {
                        if (pop(stack, &top) == '\0') {
                            return 0; 
                        }
                    }
                }
                return top == -1;
            }
            
            int main() {
                char expr[MAX];
            
                printf("Enter an expression with brackets: ");
                fgets(expr, sizeof(expr), stdin);
                expr[strcspn(expr, "\n")] = '\0';
            
                if (isBalanced(expr)) {
                    printf("The brackets are balanced.\n");
                } else {
                    printf("The brackets are not balanced.\n");
                }
            
                return 0;
            }
