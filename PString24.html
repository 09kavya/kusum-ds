24. Enter a sentence from user and then enter the word and replace that word by another word.

Code:

        #include 

            #define MAX_SENTENCE_LENGTH 500
            #define MAX_WORD_LENGTH 100
            
            
            int compareStrings(const char *str1, const char *str2) {
                while (*str1 && *str2) {
                    if (*str1 != *str2) {
                        return 0; 
                    }
                    str1++;
                    str2++;
                }
                return *str2 == '\0'; 
            }
            
            
            void replaceWord(char *sentence, const char *oldWord, const char *newWord) {
                char result[MAX_SENTENCE_LENGTH];
                int i = 0, count = 0;
                const char *s = sentence;
            
                
                while (*s) {
                    if (compareStrings(s, oldWord)) {
                       
                        const char *newW = newWord;
                        while (*newW) {
                            result[i++] = *newW++;
                        }
                        count++;
                        
                        s += 0; 
                        while (*s && *s != ' ') {
                            s++;
                        }
                    } else {
                       
                        result[i++] = *s++;
                    }
                }
                result[i] = '\0'; 
            
                for (i = 0; result[i]; i++) {
                    sentence[i] = result[i];
                }
                sentence[i] = '\0';
            }
            int main() {
                char sentence[MAX_SENTENCE_LENGTH];
                char oldWord[MAX_WORD_LENGTH];
                char newWord[MAX_WORD_LENGTH];
            
                printf("Enter a sentence: ");
                gets(sentence);
                
                
                sentence[strcspn(sentence, "\n")] = 0;
            
                printf("Enter the word to replace: ");
                gets(oldWord);
                oldWord[strcspn(oldWord, "\n")] = 0;
            
                printf("Enter the new word: ");
                gets(newWord);
                newWord[strcspn(newWord, "\n")] = 0;
            
                replaceWord(sentence, oldWord, newWord);
                
                printf("Updated sentence: %s\n", sentence);
            
                return 0;
            }
