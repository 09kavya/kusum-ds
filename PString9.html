9. Enter the string from user and copy it into another string variable.

Code:

        #include 

            int main() {
                
                char src[100]; 
                char destination[100]; 
            
                printf("Enter a string: ");
                gets(src); 
                int i = 0;
                while (src[i] != '\0') {
                    destination[i] = src[i]; 
                    i++;
                }
                destination[i] = '\0'; 
            
                printf("Copied string: %s\n", destination);
            
                return 0;
            }
