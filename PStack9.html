            #define MAX 100
            
            char stack[MAX];
            int top = -1;
            
            void push(char c) {
                if (top < MAX - 1) {
                    stack[++top] = c;
                } else {
                    printf("Stack overflow\n");
                }
            }
            
            char pop() {
                if (top >= 0) {
                    return stack[top--];
                } else {
                    printf("Stack underflow\n");
                    return '\0';
                }
            }
            
            char peek() {
                if (top >= 0) {
                    return stack[top];
                } else {
                    return '\0';
                }
            }
            int precedence(char op) {
                switch (op) {
                    case '+':
                    case '-': return 1;
                    case '*':
                    case '/': return 2;
                    case '^': return 3;
                    default: return 0;
                }
            }
            int isOperator(char c) {
                return c == '+' || c == '-' || c == '*' || c == '/' || c == '^';
            }
            
            // Function to convert infix to postfix
            void infixToPostfix(char* infix, char* postfix) {
                int j = 0;
                for (int i = 0; infix[i]; i++) {
                    if (isalnum(infix[i])) {
                        postfix[j++] = infix[i];
                    } else if (infix[i] == '(') {
                        push(infix[i]);
                    } else if (infix[i] == ')') {
                        while (top != -1 && peek() != '(') {
                            postfix[j++] = pop();
                        }
                        pop(); // pop '('
                    } else if (isOperator(infix[i])) {
                        while (top != -1 && precedence(peek()) >= precedence(infix[i])) {
                            postfix[j++] = pop();
                        }
                        push(infix[i]);
                    }
                }
                
                while (top != -1) {
                    postfix[j++] = pop();
                }
                postfix[j] = '\0';
            }
            
            // Function to reverse a string
            void reverse(char* str) {
                int n = strlen(str);
                for (int i = 0; i < n / 2; i++) {
                    char temp = str[i];
                    str[i] = str[n - i - 1];
                    str[n - i - 1] = temp;
                }
            }
            
            // Function to convert infix to prefix
            void infixToPrefix(char* infix, char* prefix) {
                reverse(infix);
                
                
                for (int i = 0; infix[i]; i++) {
                    if (infix[i] == '(') {
