4. Convert a number into binary using recursion.

Code:

        #include < stdio.h >

        void decimalToBinary(int n) {
           if (n > 1) 
               decimalToBinary(n / 2); 
                               
           printf("%d", n % 2); 
       }
                           
       int main() {
           int num;
           printf("Enter a decimal number: ");
           scanf("%d", &num);
                           
           printf("Binary representation of %d is: ", num);
           decimalToBinary(num);
           printf("\n");
                           
           return 0;
       }

Output:

        Binary representation of 10 is: 1010
