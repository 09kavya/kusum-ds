2. Create a stack implementing the operations on the characters array (push, pop, peek, isempty, isfull)

Code:

        #include 

int isFull(int i, int size) {
    return i == (size - 1);
}

int isEmpty(int i) {
    return i == -1;
}

int push(char stack[], int size, int i, char value) {
    if (isFull(i, size)) {
        printf("Stack is full.\n");
    } else {
        stack[++i] = value;
        printf("%c pushed into stack.\n", value);
    }
    return i;
}

char pop(char stack[], int *i) {
    if (isEmpty(*i)) {
        printf("Stack underflow: no element to pop.\n");
        return '\0';
    } else {
        char poppedValue = stack[*i];
        (*i)--; // Decrement the index
        printf("%c popped from the stack.\n", poppedValue);
        return poppedValue;
    }
}

char peek(char stack[], int i) {
    if (isEmpty(i)) {
        printf("\nStack is empty: no elements to peek.\n");
        return '\0'; 
    } else {
        printf("Top element is %c\n", stack[i]);
        return stack[i];
    }
}

int main() {
    int choice, size = 10, i = -1;
    char stack[10], value;

    while (1) {
        printf("1. Push\n2. Pop\n3. Peek\n4. Exit: ");
        scanf(" %d", &choice);

        switch (choice) {
            case 1:
                printf("Enter the character: ");
                scanf(" %c", &value); 
                i = push(stack, size, i, value);
                break;
            case 2:
                pop(stack, &i);
                break;
            case 3:
                peek(stack, i);
                break;
            case 4:
                return 0; 
            default:
                printf("Entered number is invalid.\n");
        }
    }
}
