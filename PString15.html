15. Enter the string from the user, enter a character also and search the character in the string.

Code:

        #include 

            int main() {
                char string[100]; 
                char searchChar;  
                int found = 0;    
            
              
                printf("Enter a string: ");
                gets(string);
                printf("Enter a character to search: ");
                scanf(" %c", &searchChar); 
            
                for (int i = 0; string[i] != '\0'; i++) 
                {
                    if (string[i] == searchChar) {
                        found = 1; // Set flag if character is found
                        printf("Character '%c' found at index %d.\n", searchChar, i);
                    }
                }
            
                if (!found) {
                    printf("Character '%c' not found in the string.\n", searchChar);
                }
            
                return 0;
            }
